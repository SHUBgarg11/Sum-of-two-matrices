Array: Sum of Elements

#include <stdio.h>
int main() {
  int arr[] = {1, 2, 3, 4};
  int sum = 0;
  for(int i = 0; i < 4; i++) sum += arr[i];
  printf("Sum = %d", sum);
  return 0;
}

String: Count Characters

#include <stdio.h>
int main() {
  char str[] = "Hello";
  int i = 0;
  while(str[i] != '\0') i++;
  printf("Length = %d", i);
  return 0;
}
